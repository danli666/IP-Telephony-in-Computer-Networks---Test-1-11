            questionElement = document.querySelector('.task_title');
        }

        if (!questionElement) {
            console.log('‚ùå –≠–ª–µ–º–µ–Ω—Ç –≤–æ–ø—Ä–æ—Å–∞ –Ω–µ –Ω–∞–π–¥–µ–Ω');
            return;
        }

        const questionText = questionElement.textContent.trim();

        // –ï—Å–ª–∏ —ç—Ç–æ –Ω–æ–≤—ã–π –≤–æ–ø—Ä–æ—Å, –Ω–∞–π–¥–µ–º –≤—Å–µ –ø—Ä–∞–≤–∏–ª—å–Ω—ã–µ –æ—Ç–≤–µ—Ç—ã
        if (!currentQuestionAnswers) {
            console.log(`üìù –ù–û–í–´–ô –í–û–ü–†–û–°: "${questionText}"`);

            const matchingQuestions = findAllMatchingQuestions(questionText);

            if (matchingQuestions.length === 0) {
                console.log('‚ùå –ü–æ–¥—Ö–æ–¥—è—â–∏–µ –≤–æ–ø—Ä–æ—Å—ã –Ω–µ –Ω–∞–π–¥–µ–Ω—ã');
                return;
            }

            // –ë–µ—Ä–µ–º –ø–µ—Ä–≤—ã–π –ø–æ–¥—Ö–æ–¥—è—â–∏–π –≤–æ–ø—Ä–æ—Å
            currentQuestionAnswers = matchingQuestions[0].answers;
            currentAnswerIndex = 0;
            selectedAnswersCount = 0;

            console.log(`‚úÖ –ü–†–ê–í–ò–õ–¨–ù–´–ï –û–¢–í–ï–¢–´ (${currentQuestionAnswers.length}):`, currentQuestionAnswers);
        }

        // –ï—Å–ª–∏ –≤—Å–µ –æ—Ç–≤–µ—Ç—ã —É–∂–µ –≤—ã–±—Ä–∞–Ω—ã
        if (currentAnswerIndex >= currentQuestionAnswers.length) {
            console.log('‚úÖ –í–°–ï –û–¢–í–ï–¢–´ –í–´–ë–†–ê–ù–´! –ú–æ–∂–µ—Ç–µ –Ω–∞–∂–∞—Ç—å "–û—Ç–≤–µ—Ç–∏—Ç—å" –≤—Ä—É—á–Ω—É—é.');
            // –°–±—Ä–∞—Å—ã–≤–∞–µ–º –¥–ª—è —Å–ª–µ–¥—É—é—â–µ–≥–æ –≤–æ–ø—Ä–æ—Å–∞
            currentQuestionAnswers = null;
            currentAnswerIndex = 0;
            selectedAnswersCount = 0;
            return;
        }

        const targetAnswer = currentQuestionAnswers[currentAnswerIndex];
        console.log(`üîç –ò–©–ï–ú –û–¢–í–ï–¢ ${currentAnswerIndex + 1}/${currentQuestionAnswers.length}: "${targetAnswer}"`);

        let answerOptions = document.querySelectorAll('.wrapper label.option');
        if (answerOptions.length === 0) {
            answerOptions = document.querySelectorAll('label.option');
        }

        if (answerOptions.length === 0) {
            console.log('‚ùå –í–∞—Ä–∏–∞–Ω—Ç—ã –æ—Ç–≤–µ—Ç–æ–≤ –Ω–µ –Ω–∞–π–¥–µ–Ω—ã');
            return;
        }

        let found = false;

        answerOptions.forEach((option, index) => {
            if (found) return;

            const rightSpan = option.querySelector('.right');
            const optionText = rightSpan ? rightSpan.textContent : option.textContent;

            console.log(`   –ü—Ä–æ–≤–µ—Ä—è–µ–º –≤–∞—Ä–∏–∞–Ω—Ç ${index + 1}: "${optionText.trim()}"`);

            if (textsMatch(optionText, targetAnswer)) {
                let input = option.querySelector('input[type="radio"], input[type="checkbox"]');
                if (input && !input.checked) {
                    input.checked = true;
                    input.dispatchEvent(new Event('change', { bubbles: true }));
                    input.dispatchEvent(new Event('click', { bubbles: true }));
                    console.log(`‚úÖ –í–´–ë–†–ê–ù –≤–∞—Ä–∏–∞–Ω—Ç ${index + 1}: "${optionText.trim()}"`);
                    found = true;
                    selectedAnswersCount++;
                    currentAnswerIndex++;
                } else if (input && input.checked) {
                    console.log(`‚ö†Ô∏è –í–∞—Ä–∏–∞–Ω—Ç ${index + 1} —É–∂–µ –≤—ã–±—Ä–∞–Ω, –ø–µ—Ä–µ—Ö–æ–¥–∏–º –∫ —Å–ª–µ–¥—É—é—â–µ–º—É`);
                    found = true;
                    currentAnswerIndex++;
                }
            }
        });

        if (!found) {
            console.log(`‚ùå –û—Ç–≤–µ—Ç "${targetAnswer}" –Ω–µ –Ω–∞–π–¥–µ–Ω, –ø–µ—Ä–µ—Ö–æ–¥–∏–º –∫ —Å–ª–µ–¥—É—é—â–µ–º—É`);
            currentAnswerIndex++;

            // –ï—Å–ª–∏ –Ω–µ –Ω–∞—à–ª–∏ —Ç–µ–∫—É—â–∏–π –æ—Ç–≤–µ—Ç, –ø—Ä–æ–±—É–µ–º —Å–ª–µ–¥—É—é—â–∏–π
            if (currentAnswerIndex < currentQuestionAnswers.length) {
                setTimeout(() => selectNextAnswer(), 500);
            }
        }

        console.log(`üìä –ü–†–û–ì–†–ï–°–°: ${selectedAnswersCount}/${currentQuestionAnswers.length} –æ—Ç–≤–µ—Ç–æ–≤ –≤—ã–±—Ä–∞–Ω–æ`);
    }

    // –û–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–ª–∞–≤–∏—à
    function handleKeyPress(event) {
        if (event.key === '2' || event.keyCode === 50) {
            if (event.target.tagName !== 'INPUT' && event.target.tagName !== 'TEXTAREA') {
                event.preventDefault();
                selectNextAnswer();
            }
        }

        // –ö–ª–∞–≤–∏—à–∞ R –¥–ª—è —Å–±—Ä–æ—Å–∞
        if (event.key === 'r' || event.key === 'R') {
            if (event.target.tagName !== 'INPUT' && event.target.tagName !== 'TEXTAREA') {
                currentQuestionAnswers = null;
                currentAnswerIndex = 0;
                selectedAnswersCount = 0;
                console.log('üîÑ –°–æ—Å—Ç–æ—è–Ω–∏–µ —Å–±—Ä–æ—à–µ–Ω–æ');
            }
        }
    }

    // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
    function init() {
        document.addEventListener('keydown', handleKeyPress);
        document.addEventListener('keypress', handleKeyPress);
        console.log('üöÄ –°–ö–†–ò–ü–¢ INTUIT AUTO COMPLETE –ó–ê–ü–£–©–ï–ù!');
        console.log('üìã –ù–∞–∂–º–∏—Ç–µ "2" –¥–ª—è –≤—ã–±–æ—Ä–∞ —Å–ª–µ–¥—É—é—â–µ–≥–æ –ø—Ä–∞–≤–∏–ª—å–Ω–æ–≥–æ –æ—Ç–≤–µ—Ç–∞');
    }

    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
    } else {
        init();
    }
