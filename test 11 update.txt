    function normalizeText(text) {
        return text.trim().toLowerCase()
            .replace(/[.,!?;:()]/g, '')
            .replace(/\s+/g, ' ')
            .trim();
    }

    // –§—É–Ω–∫—Ü–∏—è –¥–ª—è —Ç–æ—á–Ω–æ–≥–æ —Å—Ä–∞–≤–Ω–µ–Ω–∏—è —Ç–µ–∫—Å—Ç–æ–≤
    function textsMatch(optionText, correctText) {
        const normalizedOption = normalizeText(optionText);
        const normalizedCorrect = normalizeText(correctText);

        if (!normalizedOption || !normalizedCorrect) return false;

        return normalizedOption === normalizedCorrect;
    }

    function findCorrectAnswersForQuestion(questionText) {
        const normalizedQuestion = normalizeText(questionText);

        for (const [question, answers] of Object.entries(correctAnswerTexts)) {
            const normalizedDbQuestion = normalizeText(question);

            // –¢–æ—á–Ω–æ–µ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ

    }

    function updateCorrectAnswersCache() {
        const questionText = getCurrentQuestion();

        if (!questionText) {
            currentQuestionText = null;
            currentCorrectAnswers = null;
            return;
        }

        // –ï—Å–ª–∏ –≤–æ–ø—Ä–æ—Å –∏–∑–º–µ–Ω–∏–ª—Å—è
        if (questionText !== currentQuestionText) {
            console.log(`üìù –¢–µ–∫—É—â–∏–π –≤–æ–ø—Ä–æ—Å: "${questionText}"`);
            currentQuestionText = questionText;
            currentCorrectAnswers = findCorrectAnswersForQuestion(questionText);

            if (currentCorrectAnswers) {
                console.log(`‚úÖ –ù–∞–π–¥–µ–Ω—ã –ø—Ä–∞–≤–∏–ª—å–Ω—ã–µ –æ—Ç–≤–µ—Ç—ã (${currentCorrectAnswers.length}):`, currentCorrectAnswers);
            } else {
                console.log('‚ùå –ü—Ä–∞–≤–∏–ª—å–Ω—ã–µ –æ—Ç–≤–µ—Ç—ã –Ω–µ –Ω–∞–π–¥–µ–Ω—ã');
            }
        }
    }

    // –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø—Ä–æ–≤–µ—Ä–∫–∏, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ –≤–∞—Ä–∏–∞–Ω—Ç –æ—Ç–≤–µ—Ç–∞ –ø—Ä–∞–≤–∏–ª—å–Ω—ã–º
    function isOptionCorrect(optionElement) {
        if (!currentCorrectAnswers || !hoverEnabled) return false;

        const rightSpan = optionElement.querySelector('.right');
        const optionText = rightSpan ? rightSpan.textContent : optionElement.textContent;

        return currentCorrectAnswers.some(correctAnswer =>
            textsMatch(optionText, correctAnswer)
        );
    }

    // –§—É–Ω–∫—Ü–∏—è –¥–ª—è –≤—ã–±–æ—Ä–∞ –≤–∞—Ä–∏–∞–Ω—Ç–∞ –æ—Ç–≤–µ—Ç–∞
    function selectOption(optionElement) {
        if (!hoverEnabled) return;

        const input = optionElement.querySelector('input[type="radio"], input[type="checkbox"]');
        if (!input) return;

        if (!input.checked) {
            input.checked = true;
            input.dispatchEvent(new Event('change', { bubbles: true }));
            input.dispatchEvent(new Event('click', { bubbles: true }));

            console.log(`‚úÖ –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ –≤—ã–±—Ä–∞–Ω: "${optionElement.textContent.trim()}"`);
        }
    }

    // –û–±—Ä–∞–±–æ—Ç—á–∏–∫ –Ω–∞–≤–µ–¥–µ–Ω–∏—è –º—ã—à–∏
    function handleMouseOver(event) {
        updateCorrectAnswersCache();

        let target = event.target;

        // –ò—â–µ–º —Ä–æ–¥–∏—Ç–µ–ª—å—Å–∫–∏–π —ç–ª–µ–º–µ–Ω—Ç —Å –∫–ª–∞—Å—Å–æ–º option
        while (target && !target.classList?.contains('option')) {
            target = target.parentElement;
            if (!target || target === document.body) return;
        }

        if (!target || !target.classList.contains('option')) return;

        // –ü—Ä–æ–≤–µ—Ä—è–µ–º, –ø—Ä–∞–≤–∏–ª—å–Ω—ã–π –ª–∏ —ç—Ç–æ –æ—Ç–≤–µ—Ç
        if (isOptionCorrect(target)) {
            currentHoverElement = target;

            // –ó–∞–ø—É—Å–∫–∞–µ–º —Ç–∞–π–º–µ—Ä —Å –∑–∞–¥–µ—Ä–∂–∫–æ–π 500ms
            hoverTimer = setTimeout(() => {
                if (currentHoverElement === target) {
                    selectOption(target);
                    currentHoverElement = null;
                }
            }, 500); // 0.5 —Å–µ–∫—É–Ω–¥—ã –∑–∞–¥–µ—Ä–∂–∫–∏
        }
    }

    // –û–±—Ä–∞–±–æ—Ç—á–∏–∫ —É—Ö–æ–¥–∞ –º—ã—à–∏ —Å —ç–ª–µ–º–µ–Ω—Ç–∞
    function handleMouseOut(event) {
        let target = event.target;

        // –ò—â–µ–º —Ä–æ–¥–∏—Ç–µ–ª—å—Å–∫–∏–π —ç–ª–µ–º–µ–Ω—Ç —Å –∫–ª–∞—Å—Å–æ–º option
        while (target && !target.classList?.contains('option')) {
            target = target.parentElement;
            if (!target || target === document.body) return;
        }

        if (!target || !target.classList.contains('option')) return;

        // –ï—Å–ª–∏ —É—à–ª–∏ —Å —ç–ª–µ–º–µ–Ω—Ç–∞, –∫–æ—Ç–æ—Ä—ã–π –±—ã–ª –Ω–∞–≤–µ–¥–µ–Ω, –æ—Ç–º–µ–Ω—è–µ–º —Ç–∞–π–º–µ—Ä
        if (target === currentHoverElement) {
            clearTimeout(hoverTimer);
            hoverTimer = null;
            currentHoverElement = null;
        }
    }

    // –û–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–ª–∞–≤–∏—à –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏
    function handleKeyPress(event) {
        // Ctrl+Shift+D –¥–ª—è –≤—ã–≤–æ–¥–∞ –æ—Ç–ª–∞–¥–æ—á–Ω–æ–π –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏
        if (event.ctrlKey && event.shiftKey && event.key === 'D') {
            console.log('=== –û–¢–õ–ê–î–û–ß–ù–ê–Ø –ò–ù–§–û–†–ú–ê–¶–ò–Ø ===');
            console.log('–¢–µ–∫—É—â–∏–π –≤–æ–ø—Ä–æ—Å:', currentQuestionText);
            console.log('–ü—Ä–∞–≤–∏–ª—å–Ω—ã–µ –æ—Ç–≤–µ—Ç—ã:', currentCorrectAnswers);
            console.log('–ê–≤—Ç–æ–≤—ã–±–æ—Ä –≤–∫–ª—é—á–µ–Ω:', hoverEnabled);
        }

        // Ctrl+Shift+T –¥–ª—è –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏—è —Ä–µ–∂–∏–º–∞
        if (event.ctrlKey && event.shiftKey && event.key === 'T') {
            hoverEnabled = !hoverEnabled;
            console.log(hoverEnabled ? '‚úÖ –ê–≤—Ç–æ–≤—ã–±–æ—Ä –≤–∫–ª—é—á–µ–Ω' : '‚ùå –ê–≤—Ç–æ–≤—ã–±–æ—Ä –≤—ã–∫–ª—é—á–µ–Ω');
        }
    }

    // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
    function init() {
        // –î–æ–±–∞–≤–ª—è–µ–º –æ–±—Ä–∞–±–æ—Ç—á–∏–∫ –Ω–∞–≤–µ–¥–µ–Ω–∏—è –º—ã—à–∏
        document.addEventListener('mouseover', handleMouseOver);
        document.addEventListener('mouseout', handleMouseOut);

        // –î–æ–±–∞–≤–ª—è–µ–º –æ–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–ª–∞–≤–∏—à –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏
        document.addEventListener('keydown', handleKeyPress);

        // –û–±–Ω–æ–≤–ª—è–µ–º –∫—ç—à –ø—Ä–∏ –∫–ª–∏–∫–µ (–¥–ª—è –ø–µ—Ä–µ—Ö–æ–¥–∞ –∫ —Å–ª–µ–¥—É—é—â–µ–º—É –≤–æ–ø—Ä–æ—Å—É)
        document.addEventListener('click', updateCorrectAnswersCache);

        console.log('üöÄ –°–ö–†–ò–ü–¢ –ê–í–¢–û–í–´–ë–û–†–ê –ó–ê–ü–£–©–ï–ù!');
        console.log('üéØ –ù–∞–≤–µ–¥–∏—Ç–µ –∫—É—Ä—Å–æ—Ä –Ω–∞ –ø—Ä–∞–≤–∏–ª—å–Ω—ã–π –æ—Ç–≤–µ—Ç –¥–ª—è –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–≥–æ –≤—ã–±–æ—Ä–∞');
        console.log('‚è±Ô∏è  –ê–≤—Ç–æ–≤—ã–±–æ—Ä –ø—Ä–æ–∏–∑–æ–π–¥–µ—Ç —á–µ—Ä–µ–∑ 0.5 —Å–µ–∫—É–Ω–¥—ã –ø–æ—Å–ª–µ –Ω–∞–≤–µ–¥–µ–Ω–∏—è');
        console.log('üîÑ –ù–∞–∂–º–∏—Ç–µ Ctrl+Shift+T –¥–ª—è –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏—è —Ä–µ–∂–∏–º–∞ –∞–≤—Ç–æ–≤—ã–±–æ—Ä–∞');
        console.log('üìä –ù–∞–∂–º–∏—Ç–µ Ctrl+Shift+D –¥–ª—è –æ—Ç–ª–∞–¥–æ—á–Ω–æ–π –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏');
    }

    // –ó–∞–ø—É—Å–∫ –ø—Ä–∏ –∑–∞–≥—Ä—É–∑–∫–µ —Å—Ç—Ä–∞–Ω–∏—Ü—ã
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
    } else {
        init();
    }

})();
